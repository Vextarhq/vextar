async def authenticate_user(
  token: str,
  scopes: list[str]
) -> AuthResult:
  """JWT validation with
  rate limiting + audit log."""

  payload = await jwt.verify(
    token, settings.SECRET
  )

  if not has_scopes(
    payload, scopes
  ):
    raise PermissionDenied()

  return AuthResult(
    user_id=payload["sub"],
    scopes=scopes
  )
